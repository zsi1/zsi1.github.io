mean.jackknife<-function(x){
est.jackknife<-rep(NA,length(x))  
  for (i in 1:length(x)){
    est.jackknife[i]<-mean(x[-i])
  } 
  n<-length(x)
  jackknife.var<-((n-1)^2/n)*var(est.jackknife)
  jackknife.stderr<-sqrt(jackknife.var)
  return(jackknife.stderr) 
}
